Syntax Overview

Boolean

true
false
# equivalent semantic aliases
yes
no

Numbers

2014
3.1416 # numbers can have decimal part
-46.01 # also it can be signed
0xFFF # or hexadecimal expressions prefixed with "0x"

Strings

strings literals can be unquoted
'and wrapped with quotes'
"or with double quotes"
# string escape
escape this \string \end
# and unicode characters expression as strings
\u0061\u0062\u0063\u0064\u0065\u0066\u0067

Lists

# in-line lists using the dash operator and commas
- 'hello', world, yes, 3.1416
# wrapped with brackets, end of line also matters
[ 
  'hello', world
  yes, 3.1416
]
# or you can use the double-dash operator 
# to create first level lists
--
  'hello'
  world
  yes
  3.1416

Assignment

# single value assignment
hello: world!
# in-line nested assignment
it: has: a: Cool! I really like it
# primitives types assignment
boolean: yes
string: 'oli rules!'
number: 3.141592
# or lists
list: [ 1, 2, 3 ]

Blocks

# in-line blocks expressions
hello: world!
# nested
it: has: a: 'pretty syntax'
# multi-line blocks
oli rules!:
  yes # parsed as boolean
  I agree # parsed as string
  another: 
    nested: block
  end
end
# pipe block statement
block:
  | another: value
  | hello
  | 3.1416
# blocks can have attributes
oli(language, pretty: yes, open: yes, version: 'beta'): hello
# it also supports assignment blocks with attributes
oli(language, cool: yes)
# the above is equivalent to
oli(language, cool: yes): nil

Comments

# you can use single-line comment
hello: oli! # on any part of the source
## 
  or multiple lines
##
name: 
  oli is
  ## and interpolated ##
  simple!
end

References

# you can create references with "&" operator
&hello: oli!
# and consum it with *
name: *name is simple!
# another valid reference expressions
value: *{name} or *'name'
# any type of expression in the document can consume a reference
hello: it says "hello world in *hello" # =>
# single quotes strings cannot allow interpolated data references
hello: '*hello world' # => '*hello world'
# you can create referenciable values or blocks which will
# not be present in the compilation output
value = this is a document hidden value

Expressions

# block alias with implicit reference
oli > language: hello
# you can also hide the refenrence in compilation output 
oli !> language: hello
# block inheritance
&yaml: 
  rules: yes
end
oli >> yaml:
  cool: yes
end
# or deep inheritance
oli >>> yaml:
  cool: yes
end
# and multiple inheritance
oli >> yaml >> json: cool!
