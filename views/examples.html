Markup templating

!!!
html:
  head:
    @title: Welcome to Oli!
  end
  body:
    # i'm using a reference that points to 'title'
    h1: &{title} 
    # this is a in-line comment!
    div(class: container, id: main):
      p(class: main-title another-class): This is a paragraph
      a(href: 'http://h2non.github.io/oli', title: click here!): Oli Spec
    end
  end
end

Data interchange format

# hidden values
!schema url: 'http://schema.org/'

name: The Empire State Building
description: The Empire State Building is a 102-story landmark in New York City.
image: 'http://bit.ly/1hXoBhd'
context:
  name: *{schema url}/name
  description: *{schema url}/description
  image:
    id: *{schema url}/image
    type: @id
  end
  geo: *{schema url}/geo
  latitude:
    id: *{schema url}/latitude
    type > float: 'xsd:float'
  end
  longitude:
    id: *{schema url}/longitude
    type:> float
  end
  xsd: 'http://www.w3.org/2001/XMLSchema'
end
geo:
  latitude: 40.75
  longitude: 73.98
end
