Sintax Overview

Boolean

true
yes # semantic alias to true
false
no # this one to false

Numbers

1234
12.32934 # numbers can have decimal part
-46.01 # or it can be signed

Strings

strings literals can be anquoted 
'and this one is quoted'
"and it can be also double quoted"

Blocks

# in-line blocks expressions
hello: world!
# nested
it: has: a: 'pretty sintax'
# multi-line blocks
oli rules!:
  yes # parsed as boolean
  I agree # parsed as string
end

Lists

# implicit lists ussing the comma token separator
hello: world, from, oli! 
# it can be multi line also
hello:
  world
  from
  oli!
end
# or explicit using brackets
hello: [ 
  oli, 'says', hello, 1234
]

Featured Example

# this a is a comment
name: Hello Oli!
version: 0.1.0

author:
  name: Tomas Aparicio
  email: tomas@aparicio.me
  web: 'http://tomas.aparicio.me'
  social: no # boolean primitive value
end

server deploy > zeus:
  server > host url: 'http://zeus.server.com' # strings can be quoted
  port: 9000 # number as primitive value
  retry: yes # yes is a primitive 
  # nested block
  authentication: 
    # lists can be implicit using commas
    tokens:
    end
  end
end

# extends from zeus block
server deploy >> zeus > hera:
  server: 'http://hera.server.com'
end

# in-line nested structures
hey: you: are: seeing: oli!

# blocks can have attributes
do(alias: make, type: 'user question'):
  # multi-line nested structures
  you: # implitic list
    born
    in # this is a string
    [ 80, 'th' ], ? # old, ehn?
  end
end

# void native type, that representes undefined
non existence: void

# creates a copy of hera block
deploy:> hera

# short-hand reference declaration operator
&flags: --debug --trace
# hidden variable in compilation output 
!&verbose: --verbose

# templating
command: server.sh start *{flags}
# math expressions
&fallback port: <[ *{flags} + 100 ]>

# multi-line unquoted string
log:
  /var/log/deploy.log
  /var/log/error.log
end
